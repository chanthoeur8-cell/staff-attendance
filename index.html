const SHEET_NAME = 'Attendance';

function doGet(e) {
  const action = e.parameter.action;
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sheet = ss.getSheetByName(SHEET_NAME);
  if (!sheet) {
    sheet = ss.insertSheet(SHEET_NAME);
    sheet.appendRow(['Date','Name','Role','Type','Time','LateMins']);
  }

  if (action === 'get') {
    const today = new Date().toISOString().slice(0,10);
    const data = sheet.getDataRange().getValues();
    const result = {};
    data.forEach((row, i) => {
      if (i === 0) return;
      const [date, name, role, type, time, late] = row;
      if (date === today) {
        if (!result[name]) result[name] = { morning:null, afternoon:null, checkout:null };
        if (type === 'morning')   result[name].morning   = { time, late: Number(late) };
        if (type === 'afternoon') result[name].afternoon = { time, late: Number(late) };
        if (type === 'checkout')  result[name].checkout  = time;
      }
    });
    return ContentService.createTextOutput(JSON.stringify({ ok:true, records:result }))
      .setMimeType(ContentService.MimeType.JSON);
  }

  if (action === 'reset') {
    const today = new Date().toISOString().slice(0,10);
    const data = sheet.getDataRange().getValues();
    for (let i = data.length; i >= 2; i--) {
      if (data[i-1][0] === today) sheet.deleteRow(i);
    }
    return ContentService.createTextOutput(JSON.stringify({ ok:true }))
      .setMimeType(ContentService.MimeType.JSON);
  }

  return ContentService.createTextOutput(JSON.stringify({ ok:false, error:'Unknown action' }))
    .setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  const body = JSON.parse(e.postData.contents);
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sheet = ss.getSheetByName(SHEET_NAME);
  if (!sheet) {
    sheet = ss.insertSheet(SHEET_NAME);
    sheet.appendRow(['Date','Name','Role','Type','Time','LateMins']);
  }
  const today = new Date().toISOString().slice(0,10);
  sheet.appendRow([today, body.name, body.role, body.type, body.time, body.late || 0]);
  return ContentService.createTextOutput(JSON.stringify({ ok:true }))
    .setMimeType(ContentService.MimeType.JSON);
}
